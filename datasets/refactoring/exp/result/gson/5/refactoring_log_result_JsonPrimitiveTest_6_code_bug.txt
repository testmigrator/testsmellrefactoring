import org.junit.jupiter.api.Test;
import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.CsvSource;
import static org.assertj.core.api.Assertions.assertThat;
import static org.junit.jupiter.api.Assertions.assertThrows;

class JsonPrimitiveTest {

  private JsonPrimitive json;

  @BeforeEach
  void setUp() {
    // Setup can be added here if needed for future tests
  }

  @ParameterizedTest(name = "testStringAndChar with input: {0}")
  @CsvSource({
      "'abc', true, 'a', 'abc'",
      "'z', true, 'z', 'z'",
      "'true', false, null, 'true'",
      "'' , false, null, ''"
  })
  void testStringAndChar(String value, boolean isString, Character asChar, String asString) {
    json = new JsonPrimitive(value);

    if (isString) {
      assertThat(json.isString()).as("Checking if '%s' is a string", value).isTrue();
      assertThat(json.getAsCharacter()).as("Checking if first character of '%s' is '%c'", value, asChar).isEqualTo(asChar);
      assertThat(json.getAsString()).as("Checking if '%s' equals '%s'", value, asString).isEqualTo(asString);
    } else {
      assertThat(json.isString()).as("Checking if '%s' is not a string", value).isFalse();
      assertThat(json.getAsString()).as("Checking if '%s' equals '%s'", value, asString).isEqualTo(asString);
    }
  }

  @Test
  void testEmptyStringGetAsCharacter() {
    JsonPrimitive emptyString = new JsonPrimitive("");
    var e = assertThrows(UnsupportedOperationException.class, () -> emptyString.getAsCharacter());
    assertThat(e).as("Checking exception message for empty string getAsCharacter").hasMessageThat().isEqualTo("String value is empty");
  }
}
