import static org.junit.jupiter.api.Assertions.*;
import static org.hamcrest.MatcherAssert.assertThat;
import static org.hamcrest.Matchers.*;

import com.google.gson.stream.JsonReader;
import com.google.gson.stream.MalformedJsonException;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.CsvSource;

import java.io.IOException;
import java.io.StringReader;

public class JsonReaderTest {

  private JsonReader reader(String json) {
    return new JsonReader(new StringReader(json));
  }

  @Test
  public void testNestingLimit_SetAndGet() throws IOException {
    JsonReader reader = reader("[{\"a\":1}]");
    reader.setNestingLimit(2);
    assertThat(reader.getNestingLimit(), equalTo(2), "Nesting limit should be set to 2");
  }

  @Test
  public void testNestingLimit_ValidArrayAndObject() throws IOException {
    JsonReader reader = reader("[{\"a\":1}]");
    reader.setNestingLimit(2);
    reader.beginArray();
    reader.beginObject();
    assertThat(reader.nextName(), equalTo("a"), "Next name should be 'a'");
    assertThat(reader.nextInt(), equalTo(1), "Next integer should be 1");
    reader.endObject();
    reader.endArray();
  }

  @ParameterizedTest
  @CsvSource({
      "[{\"a\":[]}], 2, a, beginArray",
      "[], 0, $, beginArray",
      "[], 0, $, skipValue"
  })
  public void testNestingLimit_ExceedsLimit(String json, int limit, String path, String method) throws IOException {
    JsonReader reader = reader(json);
    reader.setNestingLimit(limit);
    reader.beginArray();
    reader.beginObject();
    assertThat(reader.nextName(), equalTo("a"), "Next name should be 'a'");
    MalformedJsonException e = assertThrows(MalformedJsonException.class, () -> {
      if ("beginArray".equals(method)) {
        reader.beginArray();
      } else if ("skipValue".equals(method)) {
        reader.skipValue();
      }
    });
    assertThat(e.getMessage(), containsString("Nesting limit " + limit + " reached at line 1 column 8 path $" + path),
        "Exception message should indicate nesting limit exceeded");
  }

  @Test
  public void testNestingLimit_TopLevelValue() throws IOException {
    JsonReader reader = reader("1");
    reader.setNestingLimit(0);
    assertThat(reader.nextInt(), equalTo(1), "Reading non-array or non-object value should be allowed");
  }

  @Test
  public void testNestingLimit_MultipleTopLevelArrays() throws IOException {
    JsonReader reader = reader("[] [[]]");
    reader.setStrictness(JsonReader.Strictness.LENIENT);
    reader.setNestingLimit(1);
    reader.beginArray();
    reader.endArray();
    reader.beginArray();
    MalformedJsonException e = assertThrows(MalformedJsonException.class, () -> reader.beginArray());
    assertThat(e.getMessage(), containsString("Nesting limit 1 reached at line 1 column 6 path $[0]"),
        "Exception message should indicate nesting limit exceeded");
  }

  @Test
  public void testNestingLimit_InvalidLimit() {
    JsonReader reader = reader("[]");
    IllegalArgumentException e = assertThrows(IllegalArgumentException.class, () -> reader.setNestingLimit(-1));
    assertThat(e.getMessage(), equalTo("Invalid nesting limit: -1"), "Exception message should indicate invalid nesting limit");
  }
}
