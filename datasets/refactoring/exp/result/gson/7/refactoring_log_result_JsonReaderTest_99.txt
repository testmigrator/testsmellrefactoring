import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.CsvSource;

import java.io.IOException;
import java.io.StringReader;

import static com.google.common.truth.Truth.assertThat;
import com.google.gson.stream.JsonReader;
import com.google.gson.stream.JsonToken;

public class JsonReaderTest {

  private JsonReader reader;

  @BeforeEach
  public void setUp(String input) throws IOException {
    reader = new JsonReader(new StringReader(input));
  }

  @ParameterizedTest
  @CsvSource({
    "true, true",
    "false, false"
  })
  public void testBooleanValues(String input, boolean expected) throws IOException {
    setUp(input);
    assertThat(reader.nextBoolean(), "Expected boolean value to be " + expected).isEqualTo(expected);
    assertThat(reader.peek(), "Expected next token to be END_DOCUMENT").isEqualTo(JsonToken.END_DOCUMENT);
  }

  @Test
  public void testNullValue() throws IOException {
    setUp("null");
    assertThat(reader.peek(), "Expected next token to be NULL").isEqualTo(JsonToken.NULL);
    reader.nextNull();
    assertThat(reader.peek(), "Expected next token to be END_DOCUMENT").isEqualTo(JsonToken.END_DOCUMENT);
  }

  @ParameterizedTest
  @CsvSource({
    "123, 123",
    "123.4, 123.4"
  })
  public void testNumericValues(String input, Number expected) throws IOException {
    setUp(input);
    if (expected instanceof Integer) {
      assertThat(reader.nextInt(), "Expected integer value to be " + expected).isEqualTo(expected.intValue());
    } else if (expected instanceof Double) {
      assertThat(reader.nextDouble(), "Expected double value to be " + expected).isEqualTo(expected.doubleValue());
    }
    assertThat(reader.peek(), "Expected next token to be END_DOCUMENT").isEqualTo(JsonToken.END_DOCUMENT);
  }

  @Test
  public void testStringValue() throws IOException {
    setUp("\"a\"");
    assertThat(reader.nextString(), "Expected string value to be a").isEqualTo("a");
    assertThat(reader.peek(), "Expected next token to be END_DOCUMENT").isEqualTo(JsonToken.END_DOCUMENT);
  }
}
