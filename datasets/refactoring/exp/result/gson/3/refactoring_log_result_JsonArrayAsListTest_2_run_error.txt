import org.junit.jupiter.params.ParameterizedTest;
import org.junit.jupiter.params.provider.CsvSource;
import static org.assertj.core.api.Assertions.assertThat;

class JsonArrayTest {

    @ParameterizedTest
    @CsvSource({
        "1, 1",
        "1, 2"
    })
    public void testSize(int initialElement, int finalSize) {
        JsonArray a = new JsonArray();
        a.add(initialElement);

        List<JsonElement> list = a.asList();
        assertThat(list).hasSize(1, "List should have size 1 after adding the first element");
        list.add(new JsonPrimitive(finalSize));
        assertThat(list).hasSize(finalSize, "List should have the final size after adding the second element");
    }
}
