class FunctionUtilsTest {

    private static final double DELTA = 1.0e-15;

    @ParameterizedTest
    @CsvSource({
        "-1.0, -0.8414709848078965, 0.5403023058681398, -0.9899924966004454",
        "-0.99, -0.838770805773882, 0.5446380920035316, -0.9898461455517415",
        // ... (additional values for t from -1.0 to 1.0 with step 0.01)
        "0.99, 0.838770805773882, 0.5446380920035316, 0.9898461455517415",
        "1.0, 0.8414709848078965, 0.5403023058681398, 0.9899924966004454"
    })
    public void testToDifferentiableUnivariate(double t, double sinT, double cosT, double secondDerivative) {
        final UnivariateFunction f0 = x -> x * x;
        final UnivariateFunction f1 = x -> 2 * x;
        final UnivariateFunction f2 = x -> 2;
        final UnivariateDifferentiableFunction f = FunctionUtils.toDifferentiable(f0, f1, f2);

        DerivativeStructure dsT = new DerivativeStructure(1, 2, 0, t);
        DerivativeStructure y = f.value(dsT.sin());

        assertEquals(sinT * sinT, f.value(sinT), DELTA, "Value at t should match expected value");
        assertEquals(sinT * sinT, y.getValue(), DELTA, "Value of DerivativeStructure should match expected value");
        assertEquals(2 * cosT * sinT, y.getPartialDerivative(1), DELTA, "First derivative should match expected value");
        assertEquals(2 * (1 - 2 * sinT * sinT), y.getPartialDerivative(2), DELTA, "Second derivative should match expected value");
    }
