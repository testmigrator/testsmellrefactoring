import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
import static org.junit.jupiter.api.Assertions.assertEquals;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class ElementTest {

    private Document doc;
    private Element table;

    @BeforeEach
    public void setupTestPrependRowToTable() {
        doc = Jsoup.parse("<table><tr><td>1</td></tr></table>");
        table = doc.select("tbody").first();
    }

    @Test
    public void testPrependRowToTable_htmlStructure() {
        table.prepend("<tr><td>2</td></tr>");
        assertEquals("<table><tbody><tr><td>2</td></tr><tr><td>1</td></tr></tbody></table>",
                TextUtil.stripNewlines(doc.body().html()), "HTML structure after prepending row should match expected output");
    }

    @Test
    public void testPrependRowToTable_siblingIndex() {
        table.prepend("<tr><td>2</td></tr>");
        Elements rows = doc.select("tr");
        for (int i = 0; i < rows.size(); i++) {
            assertEquals(i, rows.get(i).siblingIndex(), "Sibling index of row at position " + i + " should be " + i);
        }
    }
}
