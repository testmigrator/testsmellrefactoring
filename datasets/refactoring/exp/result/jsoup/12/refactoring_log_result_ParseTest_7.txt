import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.io.TempDir;
import java.io.File;
import java.io.IOException;
import static org.junit.jupiter.api.Assertions.*;
import static org.jsoup.Jsoup.parse;

public class Html5CharsetTest {

    @TempDir
    public File tempDir;

    @Test
    public void testMetaCharsetWorks() throws IOException {
        File in = new File(tempDir, "meta-charset-1.html");
        assertTrue(in.exists(), "File meta-charset-1.html does not exist");
        Document doc = parse(in, null, "http://example.com/");
        assertEquals("新", doc.text(), "Text content should be '新'");
        assertEquals("GB2312", doc.outputSettings().charset().displayName(), "Charset should be GB2312");
    }

    @Test
    public void testNoCharsetFallsBackToUtf8Incorrectly() throws IOException {
        File in = new File(tempDir, "meta-charset-2.html");
        assertTrue(in.exists(), "File meta-charset-2.html does not exist");
        Document doc = parse(in, null, "http://example.com");
        assertEquals("UTF-8", doc.outputSettings().charset().displayName(), "Charset should fall back to UTF-8");
        assertFalse("新".equals(doc.text()), "Text content should not be '新'");
    }

    @Test
    public void testFallbackToUtf8() throws IOException {
        File in = new File(tempDir, "meta-charset-3.html");
        assertTrue(in.exists(), "File meta-charset-3.html does not exist");
        Document doc = parse(in, null, "http://example.com/");
        assertEquals("UTF-8", doc.outputSettings().charset().displayName(), "Charset should be UTF-8");
        assertEquals("新", doc.text(), "Text content should be '新'");
    }
}
