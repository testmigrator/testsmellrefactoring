import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.select.Elements;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.io.TempDir;

import java.io.File;
import java.io.IOException;

import static org.junit.jupiter.api.Assertions.*;

@Test
public void testGoogleSearchIpod(@TempDir File tempDir) throws IOException {
    File in = new File(tempDir, "google-ipod.html");
    if (!in.exists() || in.isDirectory()) {
        fail("Input file does not exist or is not of the expected type.");
    }
    
    Document doc = Jsoup.parse(in, "UTF-8", "http://www.google.com/search?hl=en&q=ipod&aq=f&oq=&aqi=g10");
    assertEquals("ipod - Google Search", doc.title(), "Title of the document should match the expected value");
    
    Elements results = doc.select("h3.r > a");
    int expectedResultsCount = 12;
    assertEquals(expectedResultsCount, results.size(), "Number of search results should be " + expectedResultsCount);
    
    String firstResultUrl = "http://news.google.com/news?hl=en&q=ipod&um=1&ie=UTF-8&ei=uYlKS4SbBoGg6gPf-5XXCw&sa=X&oi=news_group&ct=title&resnum=1&ved=0CCIQsQQwAA";
    assertEquals(firstResultUrl, results.get(0).attr("href"), "First result URL should match the expected value");
    
    String secondResultUrl = "http://www.apple.com/itunes/";
    assertEquals(secondResultUrl, results.get(1).attr("href"), "Second result URL should match the expected value");
}
